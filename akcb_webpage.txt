<script>
document.addEventListener('DOMContentLoaded', function() {
  const container = document.getElementById('ai-chatbot-container');
  const widget = document.getElementById('ai-chatbot-widget');
  const toggle = document.getElementById('ai-chatbot-toggle');
  const badge = document.getElementById('ai-chatbot-badge');
  const tip = document.getElementById('ai-chatbot-tip');
  const chatIcon = document.getElementById('chat-icon');
  const closeIcon = document.getElementById('close-icon');
  
  let isOpen = false;
  let unreadCount = 0;
  let hasInteracted = false;
  
  // Show welcome tip after delay
  setTimeout(() => {
    if (!hasInteracted) {
      tip.style.display = 'block';
      setTimeout(() => {
        if (tip.style.display === 'block') {
          tip.style.display = 'none';
        }
      }, 10000);
    }
  }, 3000);
  
  // Toggle widget
  toggle.addEventListener('click', function() {
    hasInteracted = true;
    tip.style.display = 'none';
    
    isOpen = !isOpen;
    
    if (isOpen) {
      widget.classList.add('show');
      chatIcon.style.display = 'none';
      closeIcon.style.display = 'block';
      badge.style.display = 'none';
      unreadCount = 0;
      
      // Notify iframe about visibility
      widget.contentWindow?.postMessage({
        type: 'PARENT_VISIBILITY_CHANGE',
        visible: true
      }, '*');
    } else {
      widget.classList.remove('show');
      chatIcon.style.display = 'block';
      closeIcon.style.display = 'none';
      
      // Notify iframe about visibility
      widget.contentWindow?.postMessage({
        type: 'PARENT_VISIBILITY_CHANGE',
        visible: false
      }, '*');
    }
  });
  
  // Handle messages from iframe
  window.addEventListener('message', function(event) {
    if (event.origin !== 'https://ai-chatbot-1-a596.onrender.com') return;
    
    const { type, data } = event.data;
    
    switch (type) {
      case 'IFRAME_READY':
        console.log('AI Chatbot iframe ready');
        break;
        
      case 'UNREAD_MESSAGE':
        if (!isOpen && data?.count > 0) {
          unreadCount = data.count;
          badge.textContent = unreadCount > 9 ? '9+' : unreadCount;
          badge.style.display = 'flex';
        }
        break;
        
      case 'MESSAGE_READ':
        unreadCount = Math.max(0, unreadCount - 1);
        if (unreadCount === 0) {
          badge.style.display = 'none';
        } else {
          badge.textContent = unreadCount > 9 ? '9+' : unreadCount;
        }
        break;
    }
  });
  
  // Audio notification for new messages
  let audioContext;
  let hasPlayedSound = false;
  
  function initAudio() {
    if (!audioContext && !hasPlayedSound) {
      try {
        audioContext = new (window.AudioContext || window.webkitAudioContext)();
        hasPlayedSound = true;
      } catch (e) {
        console.log('Audio not supported');
      }
    }
  }
  
  function playNotificationSound() {
    if (!audioContext) return;
    
    try {
      const oscillator = audioContext.createOscillator();
      const gainNode = audioContext.createGain();
      
      oscillator.connect(gainNode);
      gainNode.connect(audioContext.destination);
      
      oscillator.frequency.setValueAtTime(800, audioContext.currentTime);
      oscillator.frequency.setValueAtTime(600, audioContext.currentTime + 0.1);
      
      gainNode.gain.setValueAtTime(0, audioContext.currentTime);
      gainNode.gain.linearRampToValueAtTime(0.1, audioContext.currentTime + 0.01);
      gainNode.gain.linearRampToValueAtTime(0, audioContext.currentTime + 0.2);
      
      oscillator.start(audioContext.currentTime);
      oscillator.stop(audioContext.currentTime + 0.2);
    } catch (e) {
      console.log('Could not play notification sound');
    }
  }
  
  // Initialize audio on first user interaction
  document.addEventListener('click', initAudio, { once: true });
  document.addEventListener('keydown', initAudio, { once: true });
  
  // Play sound for unread messages
  window.addEventListener('message', function(event) {
    if (event.origin !== 'https://ai-chatbot-1-a596.onrender.com') return;
    
    if (event.data.type === 'UNREAD_MESSAGE' && !isOpen) {
      playNotificationSound();
    }
  });
  
  // Aggressive Tawk.to removal observer
  const tawkObserver = new MutationObserver(function(mutations) {
    mutations.forEach(function(mutation) {
      mutation.addedNodes.forEach(function(node) {
        if (node.nodeType === 1) {
          // Remove Tawk.to elements
          if (node.id && (node.id.includes('tawk') || node.id.includes('tidio'))) {
            node.remove();
          }
          if (node.className && (node.className.includes('tawk') || node.className.includes('tidio'))) {
            node.remove();
          }
          // Remove script tags containing tawk.to
          if (node.tagName === 'SCRIPT' && node.src && node.src.includes('tawk.to')) {
            node.remove();
          }
          // Check children
          const tawkElements = node.querySelectorAll('[id*="tawk"], [class*="tawk"], [id*="tidio"], [class*="tidio"]');
          tawkElements.forEach(el => el.remove());
        }
      });
    });
  });
  
  tawkObserver.observe(document.body, {
    childList: true,
    subtree: true
  });
  
  // Remove any existing Tawk.to elements
  document.querySelectorAll('[id*="tawk"], [class*="tawk"], [id*="tidio"], [class*="tidio"]').forEach(el => el.remove());
});
</script>

$(document).ready(function(){

$("#carouselButton").click(function(){

$("#mycarousel").carousel('pause');

$("#mycarousel").carousel('cycle');
$("#carouselButton").children("span").removeClass('fa-play');
$("#carouselButton").children("span").addClass('fa-pause');
}
});

});
